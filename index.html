function downloadPDF() { var xhr = new XMLHttpRequest(); xhr.open('GET', '/protocol/HU-AI_Protocol_v1.0.pdf', true); xhr.responseType = 'blob'; xhr.onload = function() { if (xhr.status === 200) { var url = window.URL.createObjectURL(xhr.response); var a = document.createElement('a'); a.href = url; a.download = 'HU-AI_Protocol_v1.0.pdf'; document.body.appendChild(a); a.click(); document.body.removeChild(a); window.URL.revokeObjectURL(url); } }; xhr.send(); }
